in mlcmeltdown_server:
    1. npm init -y
    2. npm install -g nodemon
    3. npm i -D nodemon
    4. npm install -save mongoose
    5. npm i bcryptjs
    6. npm i jsonwebtoken

    "app": "nodemon app.js",
    "start": "node app.js"
